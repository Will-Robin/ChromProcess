295    def create_peak_series(self):
296        """
297        Creates arrays for series of peaks using self.clusters to identify
298        similar peaks between chromatograms. If more than one peak from a
299        single chromatogram is in a cluster, their values are added together.
300
301        The ordering of the series in the arrays is given by the order of the
302        self.clusters.
303
304        Parameters
305        ----------
306
307        Returns
308        -------
309        None
310        """
311
312        series_length = len(self.series_values)
313        number_of_peaks = len(self.clusters)
314
315        integral_series = np.zeros((series_length, number_of_peaks))
316        concentration_series = np.zeros((series_length, number_of_peaks))
317        error_series = np.zeros((series_length, number_of_peaks))
318
319        for c1, pc in enumerate(self.peak_collections):
320            is_rt = pc.internal_standard.retention_time
321
322            for c2, clust in enumerate(self.clusters):
323                for pk in pc.peaks:
324
325                    if pk.retention_time == is_rt:
326                        continue
327
328                    if pk.retention_time in clust:
329                        if pk.integral:
330                            integral_series[c1, c2] += pk.integral
331
332                        if pk.concentration:
333                            concentration_series[c1, c2] += pk.concentration
334
335                        if pk.conc_error:
336                            error_series[c1, c2] += pk.conc_error
337
338        self.integral_series = integral_series.T
339        self.concentration_series = concentration_series.T
340        self.conc_err_series = error_series.T

342    def series_traces_as_dict(self):
343        """
344        Create dictionaries of peak series values derived using the
345        self.clusters.
346
347        Parameters
348        ----------
349
350        Returns
351        -------
352        conc_dict: dict
353        err_dict: dict
354        integral_dict: dict
355        """
356
357        conc_dict = dict()
358        err_dict = dict()
359        integral_dict = dict()
360        height_dict = dict()
361
362        if len(self.cluster_assignments) == 0:
363            cluster_names = ["" for _ in self.clusters]
364        else:
365            cluster_names = [n.split(" ")[0] for n in self.cluster_assignments]
366
367        for c1, pc in enumerate(self.peak_collections):
368
369            for c2, clust in enumerate(self.clusters):
370                name = cluster_names[c2]
371                average_position = round(sum(clust) / len(clust), 3)
372
373                for pk in pc.peaks:
374                    if pk.retention_time in clust:
375                        if pk.integral:
376                            token = name + f" [{average_position}]"
377                            if token not in integral_dict:
378                                integral_dict[token] = [0.0 for _ in self.series_values]
379                            integral_dict[token][c1] += pk.integral
380
381                        if pk.height:
382                            token = name + f" [{average_position}]"
383                            if token not in height_dict:
384                                height_dict[token] = [0.0 for _ in self.series_values]
385                            height_dict[token][c1] += pk.height
386
387                        if pk.concentration:
388                            token = name + "/ M"
389                            if token not in conc_dict:
390                                conc_dict[token] = [0.0 for _ in self.series_values]
391                            conc_dict[token][c1] += pk.concentration
392
393                        if pk.conc_error:
394                            token = name + "/ M"
395                            if token not in err_dict:
396                                err_dict[token] = [0.0 for _ in self.series_values]
397                            err_dict[token][c1] += pk.conc_error
398
399        return conc_dict, err_dict, integral_dict, height_dict

def create_peak_series(self) 
296    def create_peak_series(self):
297        """
298        Creates arrays for series of peaks using self.clusters to identify
299        similar peaks between chromatograms. If more than one peak from a
300        single chromatogram is in a cluster, their values are added together.
301
302        The ordering of the series in the arrays is given by the order of the
303        self.clusters.
304
305        Parameters
306        ----------
307
308        Returns
309        -------
310        None
311        """
312
313        series_length = len(self.series_values)
314        number_of_peaks = len(self.clusters)
315
316        integral_series = np.zeros((series_length, number_of_peaks))
317        concentration_series = np.zeros((series_length, number_of_peaks))
318        error_series = np.zeros((series_length, number_of_peaks))
319
320        for c1, pc in enumerate(self.peak_collections):
321            is_rt = pc.internal_standard.retention_time
322
323            for c2, clust in enumerate(self.clusters):
324                for pk in pc.peaks:
325
326                    if pk.retention_time == is_rt:
327                        continue
328
329                    if pk.retention_time in clust:
330                        if pk.integral:
331                            integral_series[c1, c2] += pk.integral
332
333                        if pk.concentration:
334                            concentration_series[c1, c2] += pk.concentration
335
336                        if pk.conc_error:
337                            error_series[c1, c2] += pk.conc_error
338
339        self.integral_series = integral_series.T
340        self.concentration_series = concentration_series.T
341        self.conc_err_series = error_series.T
