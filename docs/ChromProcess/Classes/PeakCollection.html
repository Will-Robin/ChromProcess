 45    def remove_peaks_below_threshold(self, threshold):
 46        """
 47        Remove peaks below a certain integral threshold.
 48
 49        Note that this operates on the values held in the Peak integral
 50        attributes. If they have been normalised to an internal standard, the
 51        threshold value should probably be lower than for the 'raw' integral
 52        data.
 53
 54        Parameters
 55        ----------
 56        threshold: float
 57            Peaks below this threshold are removed.
 58
 59        Returns
 60        -------
 61        None
 62        """
 63
 64        del_idx = []
 65        for c, pk in enumerate(self.peaks):
 66            if pk.integral < threshold:
 67                del_idx.append(c)
 68
 69        self.peaks = [v for i, v in enumerate(self.peaks) if i not in del_idx]
 70
 71    def align_peaks_to_IS(self, IS_set=0.0):
 72        """
 73        Align the peak integral retention times to the internal standard's
 74        position, which will be set to IS_set.
 75
 76        Parameters
 77        ----------
 78        IS_set: float
 79            Position to set the internal standard's retention time.
 80
 81        Returns
 82        -------
 83        None
 84        """
 85
 86        is_rt = self.internal_standard.retention_time
 87        is_start = self.internal_standard.start
 88        is_end = self.internal_standard.end
 89
 90        if is_rt == 0.0:
 91            print(
 92                """Internal standard retention time is set to 0.0.
 93            No alignment performed."""
 94            )
 95        else:
 96            for p in self.peaks:
 97                p.retention_time = p.retention_time - is_rt + IS_set
 98                p.start = p.start - is_rt + IS_set
 99                p.end = p.end - is_rt + IS_set
100
101            for m in self.mass_spectra:
102                m.retention_time = m.retention_time - is_rt + IS_set
103
104            self.internal_standard.start = is_start - is_rt + IS_set
105            self.internal_standard.end = is_end - is_rt + IS_set
106            self.internal_standard.retention_time = is_rt - is_rt + IS_set

198    def apply_calibrations_to_peaks(self, calibrations, IS_conc):
199        """
200        Apply calibrations to peaks
201
202        Parameters
203        ----------
204        calibrations: ChromProcess.Classes.Instrument_Calibration
205            Container for calibration information
206        IS_conc: float
207            internal standard concentration
208
209        Returns
210        -------
211        None
212        """
213
214        if IS_conc == 0.0:
215            # results in division by 1 during conversion
216            IS_conc = 1.0
217        for pk in self.peaks:
218            if pk.assignment in calibrations.calibration_factors:
219
220                CF = calibrations.calibration_factors[pk.assignment]
221
222                if calibrations.calibration_model == "linear":
223                    pk.apply_linear_calibration(
224                        CF["B"], CF["C"], internal_standard=IS_conc
225                    )
226
227                elif calibrations.calibration_model == "quadratic":
228                    pk.apply_quadratic_calibration(
229                        CF["A"], CF["B"], CF["C"], internal_standard=IS_conc
230                    )
231                else:
232                    print("Calibration type not recognised")
