 61    def add_peaks(self, peaks: list[Peak]) -> None:
 62        """

 98    def get_mass_spectrum(self, time: float) -> tuple[np.ndarray, np.ndarray]:
 99        """
100        Get the mass spectrum at a given time point in the chromatogram.
101
102        Parameters
103        ----------
104        time: float
105            Time point for the mass spectrum in the chromatogram.
106
107        Returns
108        -------
109        m_z: numpy.ndarray[np.float64]
110            m/z value
111        intensity: numpy.ndarray[np.float64]
112            Ion counts.
113        """
114
115        inds = np.where(self.time == time)[0]
116
117        scan_inds = self.scan_indices[inds][0]
118        p_counts = self.point_counts[inds][0]
119
120        intensity = self.mz_intensity[scan_inds : scan_inds + p_counts]
121        m_z = np.round(self.mz_values[scan_inds : scan_inds + p_counts], 2)
122
123        return m_z, intensity

125    def get_ion_chromatogram(
126        self, clusters: list[list[float]]
127    ) -> dict[float, np.ndarray]:
128        """
129        Get all ion chromatograms from the Chromatogram using
130        pre-defined clusters of m/z values to bin signals.
131
132        Parameters
133        ----------
134        clusters: list
135            Clusters of m/z values. Mass values which are
136            together in the list values will be combined in
137            the output.
138
139        Returns
140        -------
141        ion_dict: dict
142            Dict of ion chromatograms
143        """
144
145        ion_dict = {np.average(c): np.zeros(len(self.time)) for c in clusters}
146        if len(self.scan_indices) != 0:
147            scan_brackets = []
148
149            for s in range(0, len(self.scan_indices) - 1):
150                scan_brackets.append([self.scan_indices[s], self.scan_indices[s + 1]])
151
152            for s, bracket in enumerate(scan_brackets):
153                st_bracket = bracket[0]
154                end_bracket = bracket[1]
155                inten = self.mz_intensity[st_bracket:end_bracket]
156                masses = self.mz_values[st_bracket:end_bracket]
157
158                for m in range(0, len(masses)):
159                    for _, c in enumerate(clusters):
160                        if masses[m] in c:
161                            ion_dict[np.average(c)][s] = inten[m]
162                            break
163
164        return ion_dict
