112    def get_mass_spectrum(self, time: float) -> tuple[np.ndarray, np.ndarray]:
113        """
114        Get the mass spectrum at a given time point in the chromatogram.
115
116        Parameters
117        ----------
118        time: float
119            Time point for the mass spectrum in the chromatogram.
120
121        Returns
122        -------
123        m_z: numpy.ndarray[np.float64]
124            m/z value
125        intensity: numpy.ndarray[np.float64]
126            Ion counts.
127        """
128
129        inds = np.where(self.time == time)[0]
130
131        scan_inds = self.scan_indices[inds][0]
132        p_counts = self.point_counts[inds][0]
133
134        intensity = self.mz_intensity[scan_inds : scan_inds + p_counts]
135        m_z = np.round(self.mz_values[scan_inds : scan_inds + p_counts], 2)
136
137        return m_z, intensity

139    def get_ion_chromatogram(
140        self, clusters: list[list[float]]
141    ) -> dict[float, np.ndarray]:
142        """
143        Get all ion chromatograms from the Chromatogram using
144        pre-defined clusters of m/z values to bin signals.
145
146        Parameters
147        ----------
148        clusters: list
149            Clusters of m/z values. Mass values which are
150            together in the list values will be combined in
151            the output.
152
153        Returns
154        -------
155        ion_dict: dict
156            Dict of ion chromatograms
157        """
158
159        ion_dict = {sum(c) / len(c): np.zeros(len(self.time)) for c in clusters}
160        if len(self.scan_indices) != 0:
161            scan_brackets = []
162
163            for s in range(0, len(self.scan_indices) - 1):
164                scan_brackets.append([self.scan_indices[s], self.scan_indices[s + 1]])
165
166            for s, bracket in enumerate(scan_brackets):
167                st_bracket = bracket[0]
168                end_bracket = bracket[1]
169                inten = self.mz_intensity[st_bracket:end_bracket]
170                masses = self.mz_values[st_bracket:end_bracket]
171
172                for m in range(0, len(masses)):
173                    for _, c in enumerate(clusters):
174                        if masses[m] in c:
175                            ion_dict[sum(c) / len(c)][s] = inten[m]
176                            break
177
178        return ion_dict
