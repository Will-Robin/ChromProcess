 10def deconvolute_region(
 11    chromatogram: Chromatogram,
 12    region: list[float],
 13    num_peaks: int = 1,
 14    baseline_subtract: bool = False,
 15    peak_bound_estimates: list[list[float]] = [],
 16) -> list[Peak]:
 17    """
 18    Deconvolute a region of a chromatogram.
 19
 20    The fitting procedure assumes that the baselinne in the selected region of
 21    the chromatogram is constant.
 22
 23    Parameters
 24    ----------
 25    chromatogram: Chromatogram
 26        Chromatogram
 27    region: list
 28        region of chromatogram under operation [lower bound, upper bound]
 29    num_peaks: int
 30        Number of peaks expected in the region.
 31    baseline_subtract: bool
 32        Whether to subtract the fitted baseline from the deconvoluted peaks
 33
 34    Returns
 35    -------
 36    new_peaks: list[Peak]
 37        List of deconvoluted peaks.
 38    """
 39
 40    ## STAGE 1 preprocessing
 41    lower, upper = (region[0], region[1])
 42
 43    # Cut out portion of signal
 44    idx = np.where((chromatogram.time > lower) & (chromatogram.time < upper))[0]
 45
 46    time = chromatogram.time[idx]
 47    signal = chromatogram.signal[idx]
 48
 49    # Find any peaks that have already been picked in the region.
 50    relevant_peaks = [
 51        chromatogram.peaks[p] for p in chromatogram.peaks if lower <= p <= upper
 52    ]
 53
 54    peaks = np.array([p.retention_time for p in relevant_peaks])
 55
 56    if len(peaks) > num_peaks:
 57        print("Deconvolution warning: deconvoluting more peaks than specified.")
 58
 59    height_init = np.array([p.height for p in relevant_peaks])
 60
 61    width_init = np.array(
 62        [np.std(chromatogram.time[p.indices], ddof=1) for p in relevant_peaks]
 63    )
 64
 65    # Pad out any extra peaks if more peaks are specified as arguments than are
 66    # found in the peak picking operation
 67    if len(peaks) < num_peaks:
 68        pad_len = num_peaks - len(peaks)
 69
 70        # First, try to fill missing peak bounds with any supplied peak bound estimates
 71        for p in peak_bound_estimates:
 72            peaks = np.hstack((peaks, [(p[1] + p[0]) / 2.0]))
 73            region_mask = (time >= p[0]) & (time <= p[1])
 74            width_init = np.hstack((width_init, [time[region_mask].std()]))
 75            height_init = np.hstack((height_init, [signal[region_mask].max()]))
 76            pad_len -= 1
 77
 78        # For any remaining deconvolutions, use estimates from the signal
 79        if pad_len > 0:
 80            peaks = np.hstack((peaks, np.linspace(time.min(), time.max(), pad_len)))
 81            width_init = np.hstack((width_init, np.full(pad_len, width_init.mean())))
 82            height_init = np.hstack((height_init, np.full(pad_len, signal.mean())))
 83
 84    # Scale heights
 85    height_init /= len(height_init)
 86
 87    baseline = signal.min()
 88
 89    ## STAGE 2: perform deconvolution.
 90    popt, pcov = deconvolution.fit_pdf(
 91        time,
 92        signal,
 93        peaks,
 94        height_init,
 95        width_init,
 96        baseline,
 97    )
 98
 99    # Format output into Peaks
100    # -> [[magnitude, position, width, baseline],]
101    result = np.vstack((np.reshape(popt[:-1], (3, -1)), np.full(len(peaks), popt[-1])))
102
103    result = result.T
104
105    # Create peak objects from deconvolution results
106    new_peaks = []
107    for x in range(0, len(result)):
108        peak_features = result[x]
109        magnitude = peak_features[0]
110        retention_time = peak_features[1]
111        width = peak_features[2]
112        baseline = peak_features[3]
113
114        start = retention_time - 4 * width  # 4
115        end = retention_time + 4 * width  # 4
116        indices = np.where((chromatogram.time >= start) & (chromatogram.time <= end))[
117            0
118        ].tolist()
119
120        # Calculate integral numerically (same as 'real' peaks).
121        time_axis = chromatogram.time[indices]
122
123        deconvoluted_signal = (
124            magnitude * norm.pdf(time_axis, loc=retention_time, scale=width) + baseline
125        )
126
127        if baseline_subtract:
128            deconvoluted_signal -= baseline
129
130        integral = np.trapz(deconvoluted_signal, x=time_axis)
131
132        deconv_params = Deconvolution(
133            {
134                "magnitude": magnitude,
135                "loc": retention_time,
136                "scale": width,
137                "baseline": baseline,
138            },
139            region,
140            "scipy.norm.pdf() + baseline",
141        )
142
143        new_peaks.append(
144            Peak(
145                retention_time,
146                start,
147                end,
148                indices=indices,
149                height=magnitude,
150                integral=integral,
151                deconvolution_params=deconv_params,
152            )
153        )
154
155    return new_peaks

def assign_retention_time( peak: ChromProcess.Classes.peak.Peak, boundaries: dict[str, list[float]]) -> str: 
 5def assign_retention_time(peak: Peak, boundaries: dict[str, list[float]]) -> str:
 6    """
 7    Takes a peak (retention times of peak) and assigns peak name based on a
 8    dictionary of boundaries. The assignment priority is based on the
 9    iteration order of the dict (Python 3 dict).
10
11    Parameters
12    ----------
13    retention_time: Peak
14        Peak to be assigned.
15    boundaries: dict
16        dictionary of boundaries for peak assignments
17
18    Returns
19    -------
20    peak_name: str
21        Name for the peak. If the peak has no assignment, the peak retention
22        time is returned, rounded to 3 decimal places.
23    """
24
25    # Default assignment to rounded retention time.
26    peak_name = str(round(peak.retention_time, 3))
27    for b in boundaries:
28        if boundaries[b][0] < peak.retention_time < boundaries[b][1]:
29            peak_name = b
30            break
31
32    return peak_name
